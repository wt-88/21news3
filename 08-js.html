document.getElementById('generate').addEventListener('click', generateCharacters); function generateCharacters() { const names = ['李寻欢', '张无忌', '令狐冲', '黄蓉', '王语嫣']; const personalities = ['豪迈', '机智', '仁慈', '狡猾', '冷酷']; const appearances = ['英俊', '潇洒', '秀丽', '端庄', '威猛']; const origins = ['少林', '武当', '峨眉', '丐帮', '明教']; const charactersContainer = document.getElementById('characters'); charactersContainer.innerHTML = ''; // 清空之前的人物信息 for (let i = 0; i < 4; i++) { // 生成四个人物 const name = names[Math.floor(Math.random() * names.length)]; const personality = personalities[Math.floor(Math.random() * personalities.length)]; const appearance = appearances[Math.floor(Math.random() * appearances.length)]; const origin = origins[Math.floor(Math.random() * origins.length)]; // 创建人物信息的HTML元素 const characterCard = document.createElement('div'); characterCard.className = 'character'; characterCard.innerHTML = `

人物${i + 1}

姓名： ${name}

性格： ${personality}

形象： ${appearance}

出身： ${origin}

`; // 将新创建的元素添加到页面中 charactersContainer.appendChild(characterCard); } }
